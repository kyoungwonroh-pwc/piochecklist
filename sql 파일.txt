-- ═══ 0. 확장 ═══
CREATE EXTENSION IF NOT EXISTS "pgcrypto";

-- ═══ 1. profiles : 사용자 프로필 ═══
CREATE TABLE IF NOT EXISTS profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  login_id      TEXT UNIQUE NOT NULL,
  full_name     TEXT NOT NULL,
  company_email TEXT UNIQUE NOT NULL,
  role          TEXT DEFAULT 'user' CHECK (role IN ('user','admin')),
  is_active     BOOLEAN DEFAULT true,
  last_login_at TIMESTAMPTZ,
  created_at    TIMESTAMPTZ DEFAULT now(),
  updated_at    TIMESTAMPTZ DEFAULT now()
);

-- ═══ 2. count_sessions : 실사 세션 ═══
CREATE TABLE IF NOT EXISTS count_sessions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_id         UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  title            TEXT NOT NULL,
  client_name      TEXT,
  status           TEXT DEFAULT 'in_progress' CHECK (status IN ('in_progress','completed','archived')),
  current_step     INT DEFAULT 0,
  mapping_json     JSONB,
  sample_size_json JSONB,
  session_template_id UUID,   -- E-2: 템플릿화된 세션 참조
  template_file_id UUID,      -- A-1/B-1: 사용한 템플릿 참조
  created_at       TIMESTAMPTZ DEFAULT now(),
  updated_at       TIMESTAMPTZ DEFAULT now()
);

-- ═══ 3. inventory_lists : RM/WIP/FG 원본 리스트 ═══
CREATE TABLE IF NOT EXISTS inventory_lists (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id  UUID NOT NULL REFERENCES count_sessions(id) ON DELETE CASCADE,
  category    TEXT NOT NULL CHECK (category IN ('rm','wip','fg')),
  file_name   TEXT,
  row_index   INT NOT NULL,
  raw_data    JSONB NOT NULL,
  item_code   TEXT,
  item_name   TEXT,
  location    TEXT,
  entity_qty  NUMERIC DEFAULT 0,
  created_at  TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_inv_session_cat ON inventory_lists(session_id, category);
CREATE INDEX IF NOT EXISTS idx_inv_search ON inventory_lists(session_id, item_code, item_name);

-- ═══ 4. count_samples : 샘플 실사 결과 ═══
CREATE TABLE IF NOT EXISTS count_samples (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id    UUID NOT NULL REFERENCES count_sessions(id) ON DELETE CASCADE,
  category      TEXT NOT NULL CHECK (category IN ('rm','wip','fg')),
  procedure     TEXT NOT NULL CHECK (procedure IN ('sheet_to_floor','floor_to_sheet')),
  inventory_id  UUID REFERENCES inventory_lists(id) ON DELETE SET NULL,
  seq           INT,
  item_code     TEXT,
  item_name     TEXT,
  spec          TEXT,
  unit          TEXT,
  location      TEXT,
  tag_no        TEXT,
  entity_qty    NUMERIC,
  pwc_qty       NUMERIC,
  notes         TEXT,
  status        TEXT DEFAULT 'pending' CHECK (status IN ('pending','match','diff')),
  confirmed     BOOLEAN DEFAULT false,
  bom_check     TEXT,
  comp_rate     TEXT,
  counted_by    UUID REFERENCES profiles(id),
  counted_at    TIMESTAMPTZ,
  device_info   TEXT,
  created_at    TIMESTAMPTZ DEFAULT now(),
  updated_at    TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_samples_session ON count_samples(session_id, category, procedure);

-- ═══ 5. template_files : PIO 템플릿 Storage 메타 ═══
CREATE TABLE IF NOT EXISTS template_files (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id     UUID REFERENCES count_sessions(id) ON DELETE SET NULL,
  storage_path   TEXT NOT NULL,
  original_name  TEXT,
  is_shared      BOOLEAN DEFAULT false,  -- A-1: true=전사공용
  uploaded_by    UUID REFERENCES profiles(id),
  created_at     TIMESTAMPTZ DEFAULT now()
);

-- ═══ 6. template_schemas : 템플릿 셀 매핑 스키마 (B-2) ═══
CREATE TABLE IF NOT EXISTS template_schemas (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  template_file_id UUID REFERENCES template_files(id) ON DELETE CASCADE,
  sheet_pattern    TEXT NOT NULL,     -- '3a', '3b', ...
  category         TEXT,              -- 'rm','wip','fg'
  procedure        TEXT,              -- 'sheet_to_floor','floor_to_sheet'
  start_row        INT NOT NULL,      -- 데이터 시작행
  col_map          JSONB NOT NULL,    -- {"itemCode":"B","itemName":"C",...}
  is_wip           BOOLEAN DEFAULT false,
  created_at       TIMESTAMPTZ DEFAULT now()
);

-- ═══ 7. mapping_history : 헤더 매핑 학습 (C-5) ═══
CREATE TABLE IF NOT EXISTS mapping_history (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source_header TEXT NOT NULL,   -- 클라이언트 리스트의 원본 컬럼명
  pio_field     TEXT NOT NULL,   -- 매핑된 PIO 필드 (itemCode, itemName, ...)
  category      TEXT,            -- 'rm','wip','fg'
  user_id       UUID REFERENCES profiles(id),
  use_count     INT DEFAULT 1,
  created_at    TIMESTAMPTZ DEFAULT now(),
  updated_at    TIMESTAMPTZ DEFAULT now(),
  UNIQUE(source_header, pio_field, category)
);
CREATE INDEX IF NOT EXISTS idx_mapping_hist ON mapping_history(source_header, category);

-- ═══ 8. session_templates : 세션 템플릿화 (E-2) ═══
CREATE TABLE IF NOT EXISTS session_templates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name             TEXT NOT NULL,      -- '삼성전자 기본 설정'
  description      TEXT,
  client_name      TEXT,
  mapping_json     JSONB,
  sample_size_json JSONB,              -- {rm:10, wip:5, fg:8}
  default_pct      NUMERIC,            -- 기본 샘플 비율(%)
  created_by       UUID REFERENCES profiles(id),
  is_public        BOOLEAN DEFAULT true,  -- 전사공용
  created_at       TIMESTAMPTZ DEFAULT now()
);

-- ═══ 9. audit_logs : 감사 로그 (C-4) ═══
CREATE TABLE IF NOT EXISTS audit_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id    UUID REFERENCES profiles(id) ON DELETE SET NULL,
  session_id UUID REFERENCES count_sessions(id) ON DELETE CASCADE,
  action     TEXT NOT NULL,     -- 'login','create_session','sample','confirm','upload_template',...
  entity     TEXT,              -- 'count_samples','count_sessions', ...
  entity_id  UUID,
  metadata   JSONB,             -- 자유 형식 부가 데이터
  ip_address TEXT,
  device_info TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_audit_user ON audit_logs(user_id, created_at DESC);
CREATE INDEX IF NOT EXISTS idx_audit_session ON audit_logs(session_id, created_at DESC);

-- ═══ 10. 트리거: 회원가입 시 profiles 자동 생성 ═══
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO profiles (id, login_id, full_name, company_email, role)
  VALUES (
    NEW.id,
    COALESCE(NEW.raw_user_meta_data->>'login_id', NEW.email),
    COALESCE(NEW.raw_user_meta_data->>'full_name', ''),
    COALESCE(NEW.raw_user_meta_data->>'company_email', NEW.email),
    CASE WHEN NEW.raw_user_meta_data->>'login_id' = 'admin' THEN 'admin' ELSE 'user' END
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();

-- updated_at 자동 갱신
CREATE OR REPLACE FUNCTION touch_updated_at()
RETURNS TRIGGER AS $$
BEGIN NEW.updated_at = now(); RETURN NEW; END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_profiles_touch ON profiles;
CREATE TRIGGER trg_profiles_touch BEFORE UPDATE ON profiles FOR EACH ROW EXECUTE FUNCTION touch_updated_at();
DROP TRIGGER IF EXISTS trg_sessions_touch ON count_sessions;
CREATE TRIGGER trg_sessions_touch BEFORE UPDATE ON count_sessions FOR EACH ROW EXECUTE FUNCTION touch_updated_at();
DROP TRIGGER IF EXISTS trg_samples_touch ON count_samples;
CREATE TRIGGER trg_samples_touch BEFORE UPDATE ON count_samples FOR EACH ROW EXECUTE FUNCTION touch_updated_at();
DROP TRIGGER IF EXISTS trg_maphist_touch ON mapping_history;
CREATE TRIGGER trg_maphist_touch BEFORE UPDATE ON mapping_history FOR EACH ROW EXECUTE FUNCTION touch_updated_at();

-- ═══ 11. is_admin 헬퍼 ═══
CREATE OR REPLACE FUNCTION is_admin()
RETURNS BOOLEAN AS $$
  SELECT EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin');
$$ LANGUAGE SQL SECURITY DEFINER STABLE;

-- ═══ 13. Storage 버킷 생성 (UI에서 생성 권장, 아래는 참고 SQL) ═══
-- Supabase Dashboard → Storage → New bucket:
--   버킷명: 'templates'           / Public: OFF
--   버킷명: 'inventory-lists'     / Public: OFF  (옵션)
-- 이후 아래 정책을 Storage > Policies > templates 에서 추가하거나 아래 SQL 실행:

DROP POLICY IF EXISTS "storage_tpl_insert" ON storage.objects;
CREATE POLICY "storage_tpl_insert" ON storage.objects FOR INSERT TO authenticated WITH CHECK (
  bucket_id = 'templates' AND (
    (storage.foldername(name))[1] = auth.uid()::text  -- 개인 폴더
    OR (storage.foldername(name))[1] = 'shared'       -- 공용 폴더 (관리자만)
  )
);
DROP POLICY IF EXISTS "storage_tpl_select" ON storage.objects;
CREATE POLICY "storage_tpl_select" ON storage.objects FOR SELECT TO authenticated USING (
  bucket_id = 'templates' AND (
    (storage.foldername(name))[1] = auth.uid()::text
    OR (storage.foldername(name))[1] = 'shared'
    OR is_admin()
  )
);
DROP POLICY IF EXISTS "storage_tpl_update" ON storage.objects;
CREATE POLICY "storage_tpl_update" ON storage.objects FOR UPDATE TO authenticated USING (
  bucket_id = 'templates' AND (
    (storage.foldername(name))[1] = auth.uid()::text OR is_admin()
  )
);
DROP POLICY IF EXISTS "storage_tpl_delete" ON storage.objects;
CREATE POLICY "storage_tpl_delete" ON storage.objects FOR DELETE TO authenticated USING (

  )

-- ═══ 14. 통계용 View (E-1 대시보드) ═══
CREATE OR REPLACE VIEW v_user_stats AS
SELECT
  p.id AS user_id,
  p.full_name, p.login_id, p.role,
  COUNT(DISTINCT s.id)                             AS session_count,
  COUNT(DISTINCT s.id) FILTER (WHERE s.status='completed') AS completed_count,
  COUNT(DISTINCT cs.id)                            AS sample_count,
  COUNT(DISTINCT cs.id) FILTER (WHERE cs.status='match')   AS match_count,
  COUNT(DISTINCT cs.id) FILTER (WHERE cs.status='diff')    AS diff_count,
  COUNT(DISTINCT cs.id) FILTER (WHERE cs.status='pending') AS pending_count,
  MAX(p.last_login_at)                             AS last_login_at
FROM profiles p
LEFT JOIN count_sessions s  ON s.owner_id = p.id
LEFT JOIN count_samples cs  ON cs.counted_by = p.id
GROUP BY p.id;

RETURNS void AS $$
  UPDATE profiles SET last_login_at = now() WHERE id = auth.uid();
$$ LANGUAGE SQL SECURITY DEFINER;

-- ═══ 16. login_id -> company_email 조회 RPC (로그인 시 사용) ═══
CREATE OR REPLACE FUNCTION get_email_by_login_id(p_login_id TEXT)
RETURNS TABLE(company_email TEXT, is_active BOOLEAN) AS $$
  SELECT company_email, is_active FROM profiles WHERE login_id = p_login_id LIMIT 1;
$$ LANGUAGE SQL SECURITY DEFINER;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;

CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER 
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
BEGIN
  INSERT INTO public.profiles (id, login_id, full_name, company_email, role)
  VALUES (
    NEW.id,
    COALESCE(NEW.raw_user_meta_data->>'login_id', split_part(NEW.email, '@', 1)),
    COALESCE(NEW.raw_user_meta_data->>'full_name', ''),
    COALESCE(NEW.raw_user_meta_data->>'company_email', NEW.email),
    CASE WHEN NEW.raw_user_meta_data->>'login_id' = 'admin' THEN 'admin' ELSE 'user' END
  );
  RETURN NEW;
EXCEPTION WHEN OTHERS THEN
  -- 에러 나도 auth 가입은 성공시킴 (profile은 수동 생성 가능)
  RAISE WARNING 'handle_new_user error: %', SQLERRM;
  RETURN NEW;
END;
$$;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();
